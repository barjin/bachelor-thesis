\newcommand\short{}

\newcommand{\emptyline}{\vspace{\baselineskip}}

\newcommand{\multilinebox}[1]{
    \begin{center}
        \framebox{
            \begin{minipage}{0.9\textwidth}
                #1
            \end{minipage}
        }
    \end{center}
}

\newcommand{\numb}[1]{
    \arabic{#1}
    \stepcounter{#1}
}

\definecolor{eclipseStrings}{RGB}{0.0,0.0,255}
\definecolor{comment}{RGB}{0,100,0}
\definecolor{eclipseKeywords}{RGB}{127,0,85}
\colorlet{numb}{magenta!60!black}

\lstdefinelanguage{json}{
    basicstyle=\normalfont\ttfamily,
    commentstyle=\color{eclipseStrings}, % style of comment
    stringstyle=\color{eclipseKeywords}, % style of strings
    showstringspaces=false,
    breaklines=true,
    string=[s]{\ \ \ "}{"},
    comment=[s]{"}{"}
}

\lstdefinelanguage{prolog}{
    basicstyle=\normalfont\ttfamily,
    commentstyle=\color{comment}, % style of comment
    stringstyle=\color{eclipseKeywords}, % style of strings
    showstringspaces=false,
    breaklines=true,
    string=[s]{"}{"},
    comment=[l]{\%},
    morecomment=[l]{:"},
}

\lstdefinelanguage{javascript}{
    basicstyle=\normalfont\ttfamily,
    commentstyle=\color{comment}, % style of comment
    stringstyle=\color{eclipseKeywords}, % style of strings
    numbers=left,
    numberstyle=\scriptsize,
    stepnumber=1,
    numbersep=8pt,
    showstringspaces=false,
    breaklines=true,
    string=[s]{"}{"},
    comment=[l]{\%},
    morecomment=[l]{:"},
    literate=
        *{0}{{{\color{numb}0}}}{1}
         {1}{{{\color{numb}1}}}{1}
         {2}{{{\color{numb}2}}}{1}
         {3}{{{\color{numb}3}}}{1}
         {4}{{{\color{numb}4}}}{1}

         {7}{{{\color{numb}7}}}{1}
         {8}{{{\color{numb}8}}}{1}
         {9}{{{\color{numb}9}}}{1}
}
